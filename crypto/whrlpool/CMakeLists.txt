CYREN_INSTALL_HEADERS(whrlpool.h PREFIX openssl)
CYREN_INSTALL_HEADERS(wp_locl.h INTERNAL)

OPENSSL_ADD_CRYPTO_SOURCES(
    wp_dgst.c
    PARENT_SCOPE)

IF(CYREN_ARCH_IS_AMD64)
    OPENSSL_ADD_CRYPTO_SOURCES(
        asm/wp-x86_64.pl
        PARENT_SCOPE PERLASM)

ELSEIF(CYREN_ARCH_IS_I386)
    OPENSSL_ADD_CRYPTO_SOURCES(
        wp_block.c
        PARENT_SCOPE)

    OPENSSL_ADD_CRYPTO_SOURCES(
        asm/wp-mmx.pl
        PARENT_SCOPE PERLASM)

ELSEIF(CYREN_ARCH_IS_PPC OR CYREN_ARCH_IS_PPC64)
    OPENSSL_ADD_CRYPTO_SOURCES(
        wp_block.c
        PARENT_SCOPE)

ELSE()
    MESSAGE(FATAL_ERROR "Unknown architecture")
ENDIF()

OPENSSL_ADD_CRYPTO_DEFINITIONS(
    -DWHIRLPOOL_ASM
    PARENT_SCOPE)

OPENSSL_ADD_CRYPTO_TESTS(
    wp_test
    PARENT_SCOPE)
